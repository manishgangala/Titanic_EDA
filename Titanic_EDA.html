for col in num_cols:
    data = df[col].dropna()
    if data.size == 0:
        continue
    plt.figure(figsize=(6,4))
    plt.hist(data, bins=30)
    plt.title(f'Histogram: {col}')
    plt.xlabel(col)
    plt.ylabel('Count')
    plt.show()
